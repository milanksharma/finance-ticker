    }


    #ticker-bar {

      width: 100%;

      height: 60px;

      align-items: center;

      overflow: hidden;

    #ticker {

<body>

  <div id="ticker-bar">

    <div id="ticker">Loading finance headlines...</div>

  </div>


  <script>

    const feedUrl = "https://api.rss2json.com/v1/api.json?rss_url=https://feeds.finance.yahoo.com/rss/2.0/headline?s=aapl,msft,goog,amzn,tsla&region=US&lang=en-US";


    async function loadFeed() {

      try {

        const res = await fetch(feedUrl);

        const data = await res.json();

        const headlines = data.items.slice(0, 10).map(i => i.title);

        document.getElementById("ticker").textContent = headlines.join(" • ");

      } catch (e) {

        document.getElementById("ticker").textContent = "Unable to load market headlines.";

        console.error(e);

      }

    }


    loadFeed();

    setInterval(loadFeed, 60000);

  </script>

</body>
